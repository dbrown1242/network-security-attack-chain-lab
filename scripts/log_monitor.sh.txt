#!/bin/bash

# ==========================================
# Real-Time Auth Log Monitor
# ==========================================

LOG_FILE="/var/log/auth.log"

if [ ! -f "$LOG_FILE" ]; then
    echo "Auth log not found."
    exit 1
fi

echo "[+] Monitoring authentication log..."
echo "Press CTRL+C to stop."

tail -f $LOG_FILE | grep --line-buffered "Failed\|Accepted"
